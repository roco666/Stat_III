\underline{Exercice 2.1}\\[0.5 cm]
Soit une variable $Z$ distribuée selon une loi normale centrée-réduite. Calculer la probabilité pour la variable $Z$ de se trouver
dans chacun des cas suivants:

 \begin{enumerate}
    \item Entre $z=0$ et $z=1.2$.\\
    \begin{eqnarray*}
    P(0<Z<1.2)&=&P(Z<1.2)-P(Z<0)\\
    &=&0.8849 -0.5 \\
    &=&0.3849
    \end{eqnarray*}
    \item Entre $z=-0.68$ et $z=0$.\\
    \begin{eqnarray*}
    P(-0.68<Z<0)&=&P(Z<0)-P(Z<-0.68)\\
    &=& P(Z<0)-(1-P(Z<0.68))\\
    &=&0.5-(1-0.7517)=0.2517
    \end{eqnarray*}
    \item Entre $z=-0.46$ et $z=2.21$.\\
    \begin{eqnarray*}
    P(-0.46<Z<2.21)&=&P(Z<2.21)-P(Z<-0.46)\\
    &=&P(Z<2.21)-(1-P(Z<0.46))\\
    &=&0.9864-(1-0.6772)=0.6636
    \end{eqnarray*}
    \item Entre $z=0.81$ et $z=1.94$.\\
    \begin{eqnarray*}
    P(0.81<Z<1.94)&=&P(Z<1.94)-P(Z<0.81)\\
    &=&0.9738-0.7910=0.1828
    \end{eqnarray*}
    \item A gauche de $z=-0.6$.\\
    \begin{eqnarray*}
     P(Z<-0.6)&=&1-P(Z<0.6)\\
     &=&1-0.7257=0.2743
    \end{eqnarray*}
    \item A droite de $z=-1.28$.\\
    \begin{eqnarray*}
    P(Z>-1.28) &=& P(Z<1.28) \\
    &=& 0.8997
    \end{eqnarray*}
    \item A droite de $z=2.05$ et à gauche de $z=-1.44$.\\
    \begin{eqnarray*}
    P(2.05 <Z<-1.44) &=& 0
    \end{eqnarray*}
     C'est impossible car $2.05>-1.44$.
     \\[0.25cm]
    \item A droite de $z=2.05$ ou à gauche de $z=-1.44$.\\
    \begin{eqnarray*}
    P(Z>2.05)+P(Z<-1.44)&=&(1-P(Z<2.05)+(1-P(Z<1.44))\\
     &=&(1-0.9798)+(1-0.9251)=0.0951
    \end{eqnarray*} 
 \end{enumerate}

\underline{Exercice 3.4}\\[0.5 cm]
a) Intervalle de confiance pour $\mu$ avec $\sigma^2$ inconnu:
\[
\mu = \bar x \pm t^{(n-1)}_{1-\frac{\alpha}{2}}\;\hat\sigma_{\bar X}
 \]
 \\[0.5cm]
 La moyenne se calcule ainsi
  \[
 \overline{x} = \frac{1}{n} \sum_{i=1}^n x_i 
 \]
 Dans le cas où les données sont groupées par classes, cette formule s'écrit 
 \[
 \overline{x} = \frac{1}{n} \sum_{i=1}^c n_i \cdot x_i 
 \] 
 où $n_i$ représente la fréquence de la classe $i$ et $c$ le nombre de classes.
\\[0.2cm]
 Nous avons donc
\[
 \overline{x} =\frac{230 \cdot 1 + 248\cdot 2 +\cdots +3\cdot 6}{688}
=\frac{1469}{688} \approx 2.1352
\]
 \\[0.25cm]
La variance de l'estimateur $\bar X$ se calcule de la manière suivante
\[
\hat\sigma_{\overline{x}} = \frac{\hat \sigma}{\sqrt{n}} 
 \]
et l'estimation de la variance de la population se calcule ainsi
\[
\hat \sigma = 
\frac{\sqrt{\sum_{i=1}^n (x_i-\overline{x})^2}}{\sqrt{n-1}}
 \]
 Dans le cas où les données sont groupées par classes, cette formule s'écrit 
 \[
\hat \sigma = 
\frac{\sqrt{\sum_{i=1}^c n_i\cdot(x_i-\overline{x})^2}}{\sqrt{n-1}}
 \]
Nous avons donc
\[
\hat \sigma = 
\frac{\sqrt{230 \cdot (1-2.1352)^2 + 248\cdot (2-2.1352)^2 +\cdots +3\cdot (6-2.1352)^2}}{\sqrt{687}}
 \approx  1.8075
 \]
et par conséquent
\[
\hat\sigma_{\overline{x}} = \frac{1.8075}{\sqrt{688}}   \approx 0.0415
 \] 
 \\[0.25cm]
 Pour $\alpha = 5\%$ et $n=688$, nous avons 
\[
t^{(n-1)}_{1-\frac{\alpha}{2}}
=
t^{(687)}_{0.975}\approx z_{0.975}\approx 1.96
\]
 \\[0.25cm]
 Nous obtenons alors
\[
\mu = \bar x \pm t^{(n-1)}_{1-\frac{\alpha}{2}}\;\hat\sigma_{\bar X}
 =2.1352\pm 0.0415 \cdot 1.96
 =  2.1352\pm 0.0813
 \]
 \\[0.25cm]
  En d'autres termes, l'intervalle de confiance pour $\alpha = 5\%$ est donné par 
 $[2.0539; 2.2165]$. 
 \newpage
 %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
b) Intervalle de confiance pour $\sigma^2$ avec $\mu$  inconnu:
 \[
 \Bigl[\frac{\sum (x_i - \overline{x})^2}{q^{(n-1)}_{1-\frac{\alpha}{2}}}

\underline{Exercice 4.6}\\[0.5 cm]

\[
\sigma_0 =\sigma_1=20
\]
\[
\mu_0 =100
\]
\[
\mu_1 =80
\]
Nous choisissons la statistique 
\[
z_0 =\frac{x-\mu_0}{\sigma_0}= \;\sim\; N(0,1)
\]
\begin{enumerate}
\item $\alpha=0.05$
\begin{eqnarray*}
x_c &=& \mu_0+ z_\alpha\cdot \sigma_0
\\&=& \mu_0+ z_{0.05}\cdot \sigma_0
\\&=&100+(-1.645)\cdot 20
\\&=&67.1
\end{eqnarray*}

\item 
$\alpha=0.01$
\begin{eqnarray*}
x_c &=& \mu_0+ z_\alpha\cdot \sigma_0
\\&=& \mu_0+ z_{0.01}\cdot \sigma_0
\\&=&100+(-2.33)\cdot 20
\\&=&53.4
\end{eqnarray*}
\item
$\alpha=0.05$\\
Le risque de seconde espèce est donné par
\begin{eqnarray*}
\beta &=& P(x_1> x_c \quad | \quad x_1 \;\sim\; N(\mu_1=80,\sigma^2_1 =20))
\\&&\\&=& P(z_1> \frac{x_c-80}{20} \quad | \quad z_1 \;\sim\; N(1,1))
\\&&\\&=& P(z_1> \frac{67.1-80}{20})
\\&&\\&=& P(z_1> -0.645)
\\&&\\&\approx& 0.74
\end{eqnarray*}

\item
$\alpha=0.01$\\
\begin{eqnarray*}
\beta &=& P(x_1> x_c \quad | \quad x_1 \;\sim\; N(\mu_1=80,\sigma^2_1 =20))
\\&&\\&=& P(z_1> \frac{x_c-80}{20} \quad | \quad z_1 \;\sim\; N(1,1))
\\&&\\&=& P(z_1> \frac{53.4-80}{20})
\\&&\\&=& P(z_1> -1.33)
\\&&\\&=& 0.9082
\end{eqnarray*}
\end{enumerate}

\begin{enumerate}

 \end{array}
\]

\item $\alpha = 0.05$
\vspace{0.1cm}
\item Test de la moyenne avec variance inconnue: 
\[
T_0=\frac{\overline{X}-\mu_0}{\hat\sigma_{\bar X}} \sim St_{n-1}
\]
\vspace{0.5cm}
\item Déterminer la région critique:
\begin{itemize}
   \item Test unilatéral à droite.\\[0.5cm]
   \item Seuil de rejet:
   \[
   r= t_{0.95}^{(50)} =1.676
   \]
\end{itemize}
\vspace{0.5cm}
\item Valeur observée de la statistique:
 \[
\hat\sigma_{\overline{x}} = 
\frac{\sqrt{\sum_{i=1}^n (x_i-\overline{x})^2}}{\sqrt{n}\cdot\sqrt{n-1}}
= \frac{\sqrt{816}}{\sqrt{51}\cdot\sqrt{50}}
=0.5657
 \]
   \[
   t_0=\frac{\overline{x}-\mu_0}{\hat\sigma_{\bar x}}
   \frac{28-27}{0.5657}=1.768
   \]
   \vspace{0.5cm}
\item $1.768> 1.676$
\\[0.5cm]
L'hypothèse $H_0$ est rejetée. En d'autres termes, l'âge moyen a augmenté.
\\[0.25 cm]
\end{enumerate}
