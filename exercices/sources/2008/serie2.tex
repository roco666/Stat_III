\section{Loi du Chi-deux} % Loi du Chi-deux
Soit la variable $Q_{10}$ suivant une loi du chi-2 à 10 degrés de liberté. Déterminer
\begin{enumerate}
	\item $P(Q_{10} < 15.98)$ 
	\item $P(Q_{10} > 18.31)$
	\item $P(Q_{10} <18.31)$
	\item $P(15.98 < Q_{10} < 18.31)$
  \item la valeur $a$ telle que $P(Q_{10} < a) = 0.975$
\end{enumerate}

\section{Loi de Student} % Loi de Student
Soit la variable $T_{8}$ suivant une loi de Student à 8 degrés de liberté. Déterminer
\begin{enumerate}
	\item $P(T_{8} < 0.546)$
	\item $P(T_{8} < - 0.546)$
	\item $ P( - 0.546 < T_{8} < 1.86)$
	\item La valeur $b$ telle que $P(T_{8}> b) = 0.6$
\end{enumerate}
