\section{Loi de Fisher-Snedecor}

\subsection{Loi de Fisher-Snedecor à $n$ et $m$ degrés de liberté}
\begin{defi}
Soit $Q_n$ et $Q_m$ deux variables aléatoires indépendantes telles que
\begin{eqnarray*}
    Q_n & \sim & \chi^2_n\\
    Q_m & \sim & \chi^2_m
\end{eqnarray*}
Alors,
$$Y_{n,m} = \frac{\;\;\frac{Q_n}{n}\;\;}{\frac{Q_m}{m}} \sim F_{n,m}$$
$F_{n,m}$ est appelée \emph{loi $F$ de Fisher-Snedecor} (ou plus simplement de Fisher) à $n$ et $m$ degrés de liberté.
\end{defi}

\begin{rem}
La loi de Fisher étant définie par le rapport de 2 quantités positives, elle ne prend donc que des valeurs positives.
\end{rem}

\begin{pro} La loi $F$ de Fisher possède les propriétés suivantes
\begin{itemize}
	\item $\E(Y_{n,m}) = \frac{m}{m-2}$, $m>2$\\
	L'espérance ne dépend pas de $n$.\\
	\item $\var (Y_{n,m}) = \frac{2m^2(m+n-2)}{n(m-2)^2(m-4)}$, $m>4$
\end{itemize}
\end{pro}

\subsection{Table de Fisher}
Pour une probabilité $p$ fixée, la table de Fisher se lit en fonction des degrés de liberté du numérateur (colonne) et du dénominateur (ligne). L'intersection de la ligne et de la colonne donne le seuil $f$ correspondant. Les tables les plus courantes sont celles correspondant à $p=95\%$ et $p=99\%$.
 
\begin{center}
\includegraphics[width=12cm]{Fisher}
\end{center}

\begin{eqnarray*}
p &=& P(Y_{n,m}\leq f)\\
\alpha &=& P(Y_{n,m}>f)
\end{eqnarray*}

\begin{ex}
$P(Y_{5,3}\leq f) = 0.95  \quad \Rightarrow \quad f=9.01$
$P(Y_{3,5}\leq f) = 0.95  \quad \Rightarrow \quad f=5.41$
\end{ex}
