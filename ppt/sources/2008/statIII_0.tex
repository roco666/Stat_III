\begin{slide}{Mesures de positionnement}
	\begin{itemize}
		\item Mode: valeur la plus fréquente
		\item Moyenne
		$$\bar{x} =\frac{x_1+\ldots +x_n}{n}=\frac{1}{n}\sum_{i=1}^{n} x_i$$
		\item Médiane\\
		50\% des observation au dessous, 50\% des observations au dessus
	\end{itemize}
\end{slide}

\begin{slide}{Mesures de dispersion}
	\begin{itemize}
		\item \'Etendue
			$$x_{[n]}-x_{[1]}$$
		\item Quantiles (exemple: quartiles)
		Le \emph{quantile d'ordre $\alpha$}, est une valeur telle que  $\alpha\%$ des données sont inférieures et $(100-\alpha)\%$ des données sont supérieures.
		\item \'Ecart type et variance
		$$\sigma^2=\frac{1}{N}\sum\limits_{i=1}^N (x_i-\mu)^2$$
		ou
		$$\sigma^2=\frac{1}{N-1}\sum\limits_{i=1}^N (x_i-\mu)^2$$
	\end{itemize}
\end{slide}

\begin{slide}{Loi de probabilités discrètes}
\begin{itemize}
  \item Probabilité \href{http://www.youtube.com/v/eY5lvKnPj_M&rel=1}{exemple}
	\item Variable aléatoire\\
	Variable qui affecte une valeur numérique à chaque issue d'une expérience aléatoire.
	\item Loi de probabilité\\
	Ensemble de toutes les valeurs de probabilité pour chaque réalisation d'une v.a.
	\item Espérance mathématique d'une loi discrète\\
	$$\E(X) = \sum_{i = 1}^{n} x_i \, \cdot \, P(X = x_i)$$
	\item Fonction de répartition d'une loi discrète\\
	$$F_X(x)=P(X\leq x)$$
\end{itemize}
\end{slide}

\begin{slide}{Loi de probabilités continues}
\begin{itemize}
	\item Une variable aléatoire continue $X$ prend ses valeurs dans un intervalle qui est un sous-ensemble de l'ensemble des nombres réels $\IR$.
  
\item
La fonction de répartition, notée $F(x)$ est définie par
$$F(a) = P(X \leq a) = \int^a_{-\infty} f(x) dx$$

\item La fonction de densité est une fonction $f$ qui vérifie
$$
\int_{-\infty}^{+\infty} f(x) dx = 1
$$
\end{itemize}
\end{slide}


\section{Inférence statistique}

\begin{slide}{Termes}
\textcolor{red}{Inférer} (lat. \it{inferre}, alléguer) : tirer comme conséquence d'un fait, d'un principe.

\begin{itemize}
	\item Une \emph{unité statistique} est le plus petit élément sur lequel
porte l'analyse statistique.
  \item Une \emph{variable statistique} est une caractéristique d'une
unité statistique.
  \item Une \emph{population} est un ensemble de toutes les unités statistiques sur lequel porte une étude
statistique.
  \item Un \emph{échantillon} est un sous-ensemble de la population.
\end{itemize}
\end{slide}


\begin{slide}[toc=]{Termes (suite)}
\begin{itemize}
	\item Un \emph{paramètre} est une mesure calculée à partir d'une population entière.
  \item Une \emph{statistique} est une mesure calculée à partir d'un échantillon.
  \item L'\emph{erreur d'échantillonnage} est la valeur de la différence entre une statistique et le paramètre évalué.
\end{itemize}
\end{slide}
